  <script>
    const noBtn = document.getElementById("noBtn");
    const yesBtn = document.getElementById("yesBtn");
    const btnRow = document.getElementById("btnRow");
    const main = document.getElementById("main");
    const yay = document.getElementById("yay");

    let dodges = 0;
    const maxDodges = 8; // after this, it stops moving so "No" is possible

    function clamp(n, min, max){ return Math.max(min, Math.min(max, n)); }

    function moveNoButton(){
      const rowRect = btnRow.getBoundingClientRect();
      const btnRect = noBtn.getBoundingClientRect();

      const padding = 6;
      const maxX = rowRect.width - btnRect.width - padding;
      const maxY = rowRect.height - btnRect.height - padding;

      // bias it to stay near the right side so it feels like it‚Äôs "escaping"
      const x = clamp(Math.random() * maxX, 0, maxX);
      const y = clamp(Math.random() * maxY, 0, maxY);

      noBtn.style.left = `${x}px`;
      noBtn.style.top = `${y}px`;
    }

    // Desktop: dodge on hover
    noBtn.addEventListener("mouseenter", () => {
      if (dodges >= maxDodges) return;
      dodges++;
      moveNoButton();

      // little escalation: make the Yes button slightly bigger as she keeps trying
      const scale = 1 + Math.min(dodges * 0.03, 0.18);
      yesBtn.style.transform = `scale(${scale})`;
    });

    // Mobile: dodge on touchstart (but still allow scrolling and not be annoying)
    noBtn.addEventListener("touchstart", (e) => {
      if (dodges >= maxDodges) return;
      e.preventDefault();
      dodges++;
      moveNoButton();
    }, { passive: false });

    yesBtn.addEventListener("click", () => {
      main.style.display = "none";
      yay.style.display = "block";
    });

    // Initialize no button position (so it‚Äôs not perfectly aligned every time)
    moveNoButton();
